    function calculateStatistics(nArray) {
    var minN = Infinity;
    var maxN = -Infinity;
    var x = sumN = meanN = 0;
    var productN = 1;
    if (i > 0) {
        for (i = 0; i < nArray.length; i++) {
        x = nArray[i];
        minN = (x < minN ? x : minN);
        maxN = (x > maxN ? x : maxN);
        sumN += x;
        productN *= x;
    }
    meanN = sumN / i;
    }
    stats = {
        'min': minN,
        'max': maxN,
        'mean': meanN,
        'sum': sumN,
        'product': productN
    };
    return stats;
} // end calculate statistics

    $('#fizzBuzzBtn').click(function () {
    var fz = Number($(fizz).val());
    var bz = Number($(buzz).val());
    var s = ss = "";
    for (j = 1; j <= 100; j++) {
        s = "";
        if (j % fz == 0) s += "fizz";
        if (j % bz == 0) s += "buzz";
        if ((j % fz != 0) && (j % bz != 0)) {
            s += j;
            if (j < 10) s += "...";
            else s += "..";
        }
        if ((j % fz == 0) && (j % bz == 0)) {
            s += ".";
            s = s.toUpperCase();
        }
        s += ".";
        if (j % 10 == 0) s += "
    ";
        ss += s;
    }
    ss += "\n";
    $('#fizzBuzzResult').html(ss);
    $('#fizzBuzzResult').css('visibility', 'visible');
}); // end fizzBuzzBtn
