    function factorial(n) {
  if (n <= 1) {
    return 1;
  } else {
    return n * factorial(n - 1);
  }

    function calculateStatistics(nArray) {
    var minN = Infinity;
    var maxN = -Infinity;

    if (i > 0) {
        for (i = 0; i < nArray.length; i++) {

        maxN = (x > maxN ? x : maxN);
        sumN += x;
        productN *= x;
    }
    meanN = sumN / i;
    }
    stats = {
        'min': minN,
        'max': maxN,

    $('#fizzBuzzBtn').click(function () {

    var s = ss = '';
    if ($('#fizz').val()=='' || $('#buzz').val()=='') {
        ss += 'Enter a number in each box.
    ';
    }
    else {
        var fz = Number($(fizz).val());
        var bz = Number($(buzz).val());
 
        if (fz < 1 || bz < 1) 
            ss += 'Numbers can\'t be less than 1.
    ';
        if (fz > 100 || bz > 100) 
            ss += 'Numbers can\'t be more than 100.
    ';
        if (fz != Math.round(fz) || bz != Math.round(bz)) 
            ss += 'Integers only, please.
    ';
        if (fz > bz) {  
            var x = fz;
            fz = bz;
            bz = x;
            $(fizz).val(fz);
            $(buzz).val(bz);
        }
    }
    if (ss == '') {
        for (j = 1; j <= 100; j++) {
            s = "";
            if (j % fz == 0) s += "fizz";
            if (j % bz == 0) s += "buzz";
            if ((j % fz != 0) && (j % bz != 0)) {
                s += j;
                if (j < 10) s += "...";
                else s += "..";
            }
            if ((j % fz == 0) && (j % bz == 0)) {
                s += ".";
                s = s.toUpperCase();
            }
            s += ".";
            if (j % 10 == 0) s += "
    ";
            ss += s;
        }
        ss += "\n";
    } else {
        ss += 'Please try again.';
    }
    $('#fizzBuzzResult').html(ss);
    $('#fizzBuzzResult').css('visibility', 'visible');
}); // end fizzBuzzBtn
